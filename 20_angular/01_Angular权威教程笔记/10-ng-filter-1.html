Short: {{now | date:'short'}}
FullDate: {{now | date:'fullDate'}}
年份格式化: {{now | date:'yyyy'}}
月份格式化: {{now | date:'MM'}}
月份英文: {{now | date:'MMMM'}}

数组过滤: {{['Ari','Lerner','Likes','To','Eat','Pizza'] | filter:'e'}}
对象过滤:{{ [{ 'name': 'Ari', 'City': 'San Francisco', 'favorite food': 'Pizza' },{ 'name': 'Nate', 'City': 'San Francisco', 'favorite food': 'indian food' }] | filter:{'favorite food': 'Pizza'} }}
自定义方法过滤：{{ ['Ari','likes','to','travel'] | filter:isCapitalized }}
Json转字符串：{{ {'name': 'Ari', 'City': 'SanFrancisco'} | json }}
字符串截取(ng1.0不支持)：{{ 'this is a test' | limitTo: 3}}
从后往前：{{ 'this is a test' | limitTo: -3}}
排序(第二个参数:true代表排序结果反转，false默认代表正序)：{{ [{ 'name': 'Ari', 'status': 'awake' },{ 'name': 'Q', 'status': 'sleeping' },{ 'name': 'Nate', 'status': 'awake' }] | orderBy:'name':true }}
